/**
 * Retrieves the form, input, and todos list elements from the DOM.
 */
const form = document.getElementById("form");
const input = document.getElementById("input");
const todosList = document.getElementById("todos");
const todos = JSON.parse(localStorage.getItem("todos"));

/**
 * Updates the local storage with the current state of the todos.
 */
const updateLocalStorage = () => {
  /**
   * Retrieves all todo elements from the DOM.
   */
  const todosElements = document.querySelectorAll("li");
  const todos = [];
  todosElements.forEach((todoElement) => {
    todos.push({
      text: todoElement.innerText,
      completed: todoElement.classList.contains("completed"),
    });
  });
  /**
   * Stringifies the todos array and sets it to local storage.
   */
  localStorage.setItem("todos", JSON.stringify(todos));
};

/**
 * Adds a new todo to the list.
 * @param {Object} todo - Optional todo object with text and completed properties.
 * @example addTodo({ text: "Buy milk", completed: true });
 */
const addTodo = (todo) => {
  let todoText = input.value;
  if (todo) todoText = todo.text;
  if (todoText) {
    const todoElement = document.createElement("li");
    if (todo && todo.completed) {
      todoElement.classList.add("completed");
    }
    todoElement.innerText = todoText;
    /**
     * Toggles the completed class on click.
     */
    todoElement.addEventListener("click", () => {
      todoElement.classList.toggle("completed");
      updateLocalStorage();
    });
    /**
     * Removes the todo on right-click.
     */
    todoElement.addEventListener("contextmenu", (e) => {
      e.preventDefault();
      todoElement.remove();
      updateLocalStorage();
    });
    todosList.appendChild(todoElement);
    input.value = "";
    updateLocalStorage();
  }
};

if (todos) {
  /**
   * Initializes the todo list by adding existing todos.
   */
  todos.forEach((todo) => addTodo(todo));
}

/**
 * Handles form submission by adding a new todo.
 */
form.addEventListener("submit", (e) => {
  e.preventDefault();
  addTodo();
});
